#想要为感兴趣的历史人物做一个简单的时间函数

        ♥先从苏轼开始吧
x<-Sys.Date()
y<-as.Date("1037-01-08")
x-y
## Time difference of 358796 days
x1<-as.numeric(format(x,"%Y"))
y1<-as.numeric(format(y,"%Y"))
print(paste("今天是苏轼诞生",x1-y1,"周年"))
## [1] "今天是苏轼诞生 982 周年"
print(paste("公元",y1,"年，即嘉祐二年，在北宋眉州眉山，一代文豪苏东坡呱呱坠地"))
## [1] "公元 1037 年，即嘉祐二年，在北宋眉州眉山，一代文豪苏东坡呱呱坠地"
